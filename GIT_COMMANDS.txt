# GitHub pe upload karne ke liye ye commands run karo:

# 1. Git repository initialize karo
git init

# 2. Sabhi files add karo
git add .

# 3. First commit karo
git commit -m "Initial commit: Rental Portal with Flask backend and Angular frontend"

# 4. GitHub pe repository banao (github.com pe jao aur new repository banao)
# Phir ye commands run karo:

# 5. Remote repository add karo (apna GitHub URL daalo)
git remote add origin https://github.com/YOUR_USERNAME/Web-of-hostel.git

# 6. Code push karo
git branch -M main
git push -u origin main
